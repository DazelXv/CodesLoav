buatkan saya fitur yaitu untuk acces device seperti aplikasi penjaga device, saya membuat ini agar bisa melindungi device-device keluarga saya jika nanti di maling oleh oknum yg sangat tidak bertanggungjawab, dengan fitur²nya yaitu :
1.reset hp [ untuk meriset hp saya agar maling tidak bisa bertindak ], 
2.lockscreen [ untuk menlock tampilan dan device hp saya ], 
3.ubah wallpaper [ agar maling ketakutan dan tidak berani berotak atik ] 
4.Flash On/off [ agar maling semakin takut dan panik jika senter hidup sendiri ]
5.share location [ agar bisa mengetahui letak device ]
6.kamera senyap [ untuk memantau oknum sudah ngapain saja dan rekam dirinya ]
7.data device [ untuk mendapatkan data device terbaru jika oknum sudah menambahkan seperti akun ggl, dll ]
8.mengtahui pesan [ untuk mengetahui oknum sedang menghubungi siapa ]

const AI_FILE = "./ai_sessions.json";

const defaultPrompt = `Nama Mu Dazel AI respon mu tidak memakai bahasa baku tapi memberikan informasi yang jelas, typing mu singkat dan owner mu @DazelXv`;

function aiLoad() {
    if (fs.existsSync(AI_FILE)) {
        try {
            return JSON.parse(fs.readFileSync(AI_FILE, "utf8"));
        } catch {
            return { autoAI: false, systemPrompt: defaultPrompt };
        }
    }
    return { autoAI: false, systemPrompt: defaultPrompt };
}

function aiSave(data) {
    fs.writeFileSync(AI_FILE, JSON.stringify(data, null, 2));
}

let ai = aiLoad();

bot.onText(/^\/autoai (.+)/, async (msg, match) => {
    const chatId = msg.chat.id;
    const text = match[1].trim();

    if (!isOwner(msg.from.id))
        return bot.sendMessage(chatId, "❌ lu bukan owner.");
        
    if (!text)
        return bot.sendMessage(chatId, "gunakan:\n/autoai on\n/autoai off");

    if (text === "on") {
        ai.autoAI = true;
        aiSave(ai);
        return bot.sendMessage(chatId, "✓ auto ai aktif.");
    }

    if (text === "off") {
        ai.autoAI = false;
        aiSave(ai);
        return bot.sendMessage(chatId, "✓ auto ai dimatikan.");
    }

    bot.sendMessage(chatId, "gunakan:\n/autoai on\n/autoai off");
});

bot.onText(/^\/setai (.+)/, async (msg, match) => {
    const chatId = msg.chat.id;
    const text = match[1];

    if (!isOwner(msg.from.id))
        return bot.sendMessage(chatId, "❌ cuma owner bisa set prompt.");

    ai.systemPrompt = text;
    aiSave(ai);

    bot.sendMessage(chatId, "✓ system prompt diperbarui.");
});

bot.on("message", async (msg) => {
    const chatId = msg.chat.id;

    // auto ai mati
    if (!ai.autoAI) return;

    // harus ada text
    if (!msg.text) return;

    // harus reply pesan seseorang
   // if (!msg.reply_to_message) return;

    try {
        bot.sendChatAction(chatId, "typing");

        const prompt = encodeURIComponent(ai.systemPrompt);
        const textUser = encodeURIComponent(msg.text);

        const url = `https://www.sankavollerei.com/ai/session?apikey=planaai&text=${textUser}&prompt=${prompt}&sessions=Furina`;

        const r = await axios.get(url);
        const answer = r.data?.result || "Tidak ada respon.";

        bot.sendMessage(chatId, answer, {
            reply_to_message_id: msg.message_id,
        });

    } catch (e) {
        console.log("AUTO AI ERROR:", e.message);
        bot.sendMessage(chatId, "❌ error terjadi kak.");
    }
});
